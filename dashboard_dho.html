/** * ============================================================ * DASHBOARD DHO — API (Google Apps Script) * ============================================================ * Este script é 100% genérico: ele não conhece "Onboarding" nem * "Treinamentos" — ele apenas lê a aba "Configurações" para saber * quais abas de dados existem e devolve cada uma como uma lista * de objetos JSON (usando a primeira linha como cabeçalho). * * Isso significa que ADICIONAR UM NOVO INDICADOR não exige tocar * neste arquivo: basta criar a aba de dados na planilha e uma * linha na aba "Configurações" apontando para ela (ver README.md * e docs/SHEETS_STRUCTURE.md para o passo a passo completo). * * ------------------------------------------------------------ * COMO INSTALAR * 1. Abra a planilha > Extensões > Apps Script. * 2. Apague o conteúdo de Code.gs e cole todo este arquivo. * 3. Implantar > Nova implantação > tipo "App da Web". * - Executar como: Eu * - Quem tem acesso: Qualquer pessoa * 4. Copie a URL gerada (termina em /exec) e cole em js/config.js. * 5. Sempre que editar este script, gere uma NOVA VERSÃO em * "Gerenciar implantações" > lápis > Nova versão > Implantar. * ============================================================ */ var SHEET_CONFIG = 'Configurações'; // ════════════════════════════════════════════════════════════ // ENTRADAS HTTP // ════════════════════════════════════════════════════════════ function doGet(e) { var action = (e.parameter && e.parameter.action) || 'snapshot'; var result; try { if (action === 'config') { result = { config: getConfigRows() }; } else { result = buildSnapshot(); } } catch (err) { result = { error: err.message }; } return respondJson(result); } function doPost(e) { var response; try { var body = JSON.parse(e.postData.contents); var action = body.action; var payload = body.payload || {}; var result; if (action === 'addTreinamento') result = addTreinamento(payload); else if (action === 'addBemEstar') result = addBemEstar(payload); else if (action === 'updateMapeamento') result = updateMapeamento(payload); else throw new Error('Ação desconhecida: ' + action); response = { ok: true, result: result }; } catch (err) { response = { ok: false, error: err.message }; } return respondJson(response); } function respondJson(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } function getSS() { return SpreadsheetApp.getActiveSpreadsheet(); } function getSheetOrThrow(name) { var sh = getSS().getSheetByName(name); if (!sh) throw new Error('Aba não encontrada: "' + name + '". Confira o nome exato na planilha.'); return sh; } // ════════════════════════════════════════════════════════════ // LEITURA GENÉRICA // ════════════════════════════════════════════════════════════ /** * Lê a aba "Configurações" e retorna um array de objetos, um por * indicador ativo. Colunas esperadas (linha 1 = cabeçalho): * Id | Nome | Categoria | Tipo | AbaDados | TipoGrafico | Icone | Cor | Ordem | Ativo */ function getConfigRows() { return sheetToObjects(SHEET_CONFIG); } /** * Converte QUALQUER aba em um array de objetos, usando a primeira * linha como nomes de campo. Datas viram string ISO (aaaa-mm-dd) * para o front-end tratar com segurança; números e texto ficam como estão. */ function sheetToObjects(sheetName) { var sh = getSheetOrThrow(sheetName); var values = sh.getDataRange().getValues(); if (values.length < 2) return []; var headers = values[0].map(function (h) { return String(h).trim(); }); var rows = []; for (var i = 1; i < values.length; i++) { var row = values[i]; if (row.every(function (c) { return c === '' || c === null; })) continue; // linha em branco var obj = {}; headers.forEach(function (header, colIndex) { if (!header) return; // ignora colunas sem cabeçalho var cell = row[colIndex]; obj[header] = normalizeCell(cell); }); rows.push(obj); } return rows; } function normalizeCell(cell) { if (Object.prototype.toString.call(cell) === '[object Date]') { // Formato ISO simples (aaaa-mm-dd) — fácil de interpretar no JS do front-end. var y = cell.getFullYear(), m = ('0' + (cell.getMonth() + 1)).slice(-2), d = ('0' + cell.getDate()).slice(-2); return y + '-' + m + '-' + d; } return cell; } /** * Monta o snapshot completo: a configuração + os dados de cada * aba referenciada por ela (sem duplicar leitura da mesma aba). */ function buildSnapshot() { var config = getConfigRows(); var data = {}; config.forEach(function (c) { var aba = c.AbaDados; if (!aba || data[aba]) return; // já lida (duas configs podem apontar pra mesma aba) try { data[aba] = sheetToObjects(aba); } catch (err) { data[aba] = []; // aba ausente não deve derrubar o snapshot inteiro console.error('Erro lendo aba "' + aba + '": ' + err.message); } }); return { generatedAt: new Date().toISOString(), config: config, data: data }; } // ════════════════════════════════════════════════════════════ // ESCRITA // ════════════════════════════════════════════════════════════ /** * Adiciona uma linha em uma aba de treinamentos, respeitando a * ORDEM DAS COLUNAS que já existir na aba (lê o cabeçalho antes * de escrever, então funciona mesmo que a ordem mude). * payload: { aba, tema, mes: "YYYY-MM", conteudo, responsavel, participantes } */ function addTreinamento(p) { if (!p.aba || !p.tema || !p.mes || !p.responsavel) throw new Error('Campos obrigatórios faltando.'); var partes = p.mes.split('-'); var dataVal = new Date(Number(partes[0]), Number(partes[1]) - 1, 1); return appendRowByHeaders(p.aba, { Tema: p.tema, Responsavel: p.responsavel, Conteudo: p.conteudo || '', Data: dataVal, Participantes: Number(p.participantes) || 0, }); } /** * payload: { aba, modalidade, data: "YYYY-MM-DD", participantes } */ function addBemEstar(p) { if (!p.aba || !p.modalidade || !p.data) throw new Error('Campos obrigatórios faltando.'); var partes = p.data.split('-'); var dataVal = new Date(Number(partes[0]), Number(partes[1]) - 1, Number(partes[2])); return appendRowByHeaders(p.aba, { Modalidade: p.modalidade, Participante: 'Registro via dashboard (' + (Number(p.participantes) || 1) + ' participante(s))', Data: dataVal, }); } /** * Atualiza mês/responsável/status de uma linha de mapeamento, * localizando-a pelo texto exato da coluna "Tema". * payload: { aba, tema, mes, responsavel, status } */ function updateMapeamento(p) { if (!p.aba || !p.tema) throw new Error('Campos obrigatórios faltando.'); var sh = getSheetOrThrow(p.aba); var values = sh.getDataRange().getValues(); var headers = values[0].map(function (h) { return String(h).trim(); }); var colTema = headers.indexOf('Tema'); var colMes = ensureColumn(sh, headers, 'MesPrevisto'); var colResp = ensureColumn(sh, headers, 'Responsavel'); var colStatus = ensureColumn(sh, headers, 'Status'); if (colTema === -1) throw new Error('A aba "' + p.aba + '" precisa ter uma coluna "Tema".'); for (var i = 1; i < values.length; i++) { if (String(values[i][colTema]).trim() === String(p.tema).trim()) { sh.getRange(i + 1, colMes + 1).setValue(p.mes || ''); sh.getRange(i + 1, colResp + 1).setValue(p.responsavel || ''); sh.getRange(i + 1, colStatus + 1).setValue(p.status || 'Planejado'); return { updated: true, row: i + 1 }; } } throw new Error('Tema não encontrado na aba "' + p.aba + '": ' + p.tema); } /** * Garante que uma coluna com o cabeçalho `name` exista na aba; * cria no final se não existir. Retorna o índice (0-based). */ function ensureColumn(sh, headers, name) { var idx = headers.indexOf(name); if (idx !== -1) return idx; var newCol = headers.length + 1; sh.getRange(1, newCol).setValue(name); headers.push(name); return newCol - 1; } /** * Adiciona uma linha numa aba respeitando a posição de cada * campo conforme o cabeçalho atual (não assume ordem fixa). * Campos do objeto `valuesByHeader` que não existirem como * coluna são ignorados silenciosamente (evita quebrar em abas * com colunas extras/customizadas pelo usuário). */ function appendRowByHeaders(sheetName, valuesByHeader) { var sh = getSheetOrThrow(sheetName); var headerRow = sh.getRange(1, 1, 1, Math.max(sh.getLastColumn(), 1)).getValues()[0] .map(function (h) { return String(h).trim(); }); var newRow = headerRow.map(function (header) { return Object.prototype.hasOwnProperty.call(valuesByHeader, header) ? valuesByHeader[header] : ''; }); sh.appendRow(newRow); return { added: true }; }
